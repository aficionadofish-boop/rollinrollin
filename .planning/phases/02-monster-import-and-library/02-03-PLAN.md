      COLUMNS = ["Name", "CR", "Type", ""]  # "" = incomplete badge column

      _cr_to_float(cr: str) -> float:
        Handles: '1/4' → 0.25, '1/2' → 0.5, '1/8' → 0.125, '0' → 0.0, '17' → 17.0,
        '?' / '' / '—' / '-' → -1.0 (sorts to top for easy spotting)
        Use try/except for the float() call, split('/') for fractions.

      class MonsterTableModel(QAbstractTableModel):
        __init__(self, monsters: list = None, parent=None): store as self._monsters
        rowCount(parent=QModelIndex()) -> int: len(self._monsters)
        columnCount(parent=QModelIndex()) -> int: 4
        data(index, role=Qt.DisplayRole):
          - Qt.DisplayRole col 0: monster.name
          - Qt.DisplayRole col 1: monster.cr
          - Qt.DisplayRole col 2: monster.creature_type
          - Qt.DisplayRole col 3: "!" if monster.incomplete else ""
          - Qt.UserRole col 1: _cr_to_float(monster.cr)  ← numeric sort key
          - All other role/col combos: return None
        headerData(section, orientation, role=Qt.DisplayRole):
          return COLUMNS[section] for horizontal + DisplayRole; else None
        reset_monsters(monsters: list) -> None:
          self.beginResetModel(); self._monsters = list(monsters); self.endResetModel()
        monster_at(row: int) -> Monster:
          return self._monsters[row]  — used by LibraryTab to retrieve selected Monster

    Create src/ui/monster_filter.py with MonsterFilterProxyModel:
      Imports: PySide6.QtCore (QSortFilterProxyModel, Qt, QModelIndex)

      class MonsterFilterProxyModel(QSortFilterProxyModel):
        __init__(self, parent=None):
          super().__init__(parent)
          self._type_filter: str = ""
          self._incomplete_only: bool = False
          self.setFilterCaseSensitivity(Qt.CaseInsensitive)
          self.setSortRole(Qt.UserRole)  # numeric CR sort

        set_type_filter(self, creature_type: str) -> None:
          self._type_filter = creature_type; self.invalidateFilter()

        set_incomplete_only(self, incomplete_only: bool) -> None:
          self._incomplete_only = incomplete_only; self.invalidateFilter()

        filterAcceptsRow(self, source_row: int, source_parent: QModelIndex) -> bool:
          model = self.sourceModel()
          monster = model._monsters[source_row]

          # Text search: AND across name, cr, creature_type
          # Use filterRegularExpression().pattern() to get the current filter string
          text = self.filterRegularExpression().pattern()
          if text:
            searchable = f"{monster.name} {monster.cr} {monster.creature_type}"
            if text.lower() not in searchable.lower():
              return False

          # Type dropdown filter (AND with text)
          if self._type_filter:
            if monster.creature_type.lower() != self._type_filter.lower():
              return False

          # Incomplete-only toggle (AND with above)
          if self._incomplete_only and not monster.incomplete:
            return False

          return True
