2. Define `_DEFAULT_STYLESHEET` — a light theme variant:
   - Background: #F5F5F5 (light gray), text: #1A1A1A (near black)
   - Input fields: #FFFFFF background, #1A1A1A text
   - Tabs: #E8E8E8 bg, selected: #F5F5F5
   - Buttons: #E0E0E0 bg, #1A1A1A text
   - Accent: #2979FF (blue) for focus borders, selections, checked buttons
   - Same structural layout as dark (same selectors, same padding/radius values)
   - Include QPushButton:checked with accent color

3. Define `_HIGH_CONTRAST_STYLESHEET` — THEME-03:
   - Background: #000000 (pure black), text: #FFFFFF (pure white)
   - Input fields: #0A0A0A background, #FFFFFF text
   - Accent: #FFD700 (gold/yellow) for maximum contrast
   - Borders: #FFFFFF (white) 2px
   - No mid-tones, no subtle grays — everything is black, white, or gold
   - Include QPushButton:checked with gold border + text

4. Define `_CUSTOM_TEMPLATE` — a template string using `{bg}`, `{text}`, `{accent}`, `{input_bg}`, `{border}` placeholders. Same structure as the dark stylesheet but with substitutable colors. `{input_bg}` is derived as slightly darker/lighter than `{bg}`. `{border}` is derived as a mid-tone between `{bg}` and `{text}`.

5. Implement `ThemeService` class:
   ```python
   class ThemeService:
       _PRESETS = {
           "dark": _DARK_STYLESHEET,
           "default": _DEFAULT_STYLESHEET,
           "high_contrast": _HIGH_CONTRAST_STYLESHEET,
       }

       def build_stylesheet(self, settings: AppSettings) -> str:
           """Build the full app stylesheet from settings."""
           has_custom = bool(settings.text_color or settings.bg_color or settings.accent_color)
           if settings.theme_name in self._PRESETS and not has_custom:
               return self._PRESETS[settings.theme_name]
           # Custom colors: format the template with user colors, falling back to preset defaults
           base = self._get_preset_defaults(settings.theme_name)
           return _CUSTOM_TEMPLATE.format(
               bg=settings.bg_color or base["bg"],
               text=settings.text_color or base["text"],
               accent=settings.accent_color or base["accent"],
               input_bg=settings.bg_color and _darken(settings.bg_color) or base["input_bg"],
               border=settings.bg_color and _midtone(settings.bg_color, settings.text_color or base["text"]) or base["border"],
           )

       def apply(self, settings: AppSettings) -> None:
           from PySide6.QtWidgets import QApplication
           stylesheet = self.build_stylesheet(settings)
           QApplication.instance().setStyleSheet(stylesheet)

       def get_accent_color(self, settings: AppSettings) -> str:
           """Return the effective accent color for the current theme."""
           if settings.accent_color:
               return settings.accent_color
           defaults = self._get_preset_defaults(settings.theme_name)
           return defaults["accent"]

       def _get_preset_defaults(self, theme_name: str) -> dict:
           defaults = {
               "dark": {"bg": "#2B2B2B", "text": "#E0E0E0", "accent": "#4DA6FF", "input_bg": "#1E1E1E", "border": "#555555"},
               "default": {"bg": "#F5F5F5", "text": "#1A1A1A", "accent": "#2979FF", "input_bg": "#FFFFFF", "border": "#CCCCCC"},
               "high_contrast": {"bg": "#000000", "text": "#FFFFFF", "accent": "#FFD700", "input_bg": "#0A0A0A", "border": "#FFFFFF"},
           }
           return defaults.get(theme_name, defaults["dark"])
   ```

6. Add helper functions `_darken(hex_color) -> str` and `_midtone(hex1, hex2) -> str` as module-level functions for deriving input_bg and border from custom colors. `_darken` shifts each RGB channel toward 0 by ~15%. `_midtone` averages the two colors.

No Qt dependencies in ThemeService except for the `apply()` method which lazy-imports QApplication. The stylesheet strings and build logic are pure Python.
